import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.IOException;
import java.nio.charset.StandardCharsets;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.nio.file.StandardOpenOption;
import java.util.ArrayList;
import java.util.LinkedHashMap;
import java.util.List;
import java.util.Map;

/**
 * CSVマージツール（単一ファイル版）。
 *
 * <p>2つのCSV（A・B）を読み込み、AのヘッダーをキーにBの列を並び替え、
 * AとBを縦結合して output.csv に出力する。
 *
 * <p>仕様：
 * <ul>
 *   <li>ヘッダー照合は大文字小文字を無視（出力はAの表記を使用）</li>
 *   <li>AにあってBにない列 → Bの該当列を空欄で補完</li>
 *   <li>BにあってAにない列 → エラー出力して処理中断（output.csvは生成しない）</li>
 *   <li>入出力ともUTF-8固定、カンマ区切り、ダブルクォート対応</li>
 *   <li>フィールド前後スペースはトリム</li>
 * </ul>
 *
 * <p>使用方法：
 * <pre>
 *   java CsvMergerApp.java &lt;CSVファイルA&gt; &lt;CSVファイルB&gt;
 * </pre>
 * <p>出力：カレントディレクトリに output.csv を生成（毎回上書き）
 */
public class CsvMergerApp {

    static final String OUTPUT_FILE_NAME = "output.csv";

    // ═════════════════════════════════════════════════════════════
    // エントリーポイント
    // ═════════════════════════════════════════════════════════════

    public static void main(String[] args) {
        // ── 引数チェック ──────────────────────────────────────────
        if (args.length != 2) {
            System.err.println("[ERROR] 引数が不正です。");
            System.err.println("使用方法: java CsvMergerApp.java <CSVファイルA> <CSVファイルB>");
            System.exit(1);
        }

        Path pathA = Paths.get(args[0]);
        Path pathB = Paths.get(args[1]);

        // ── ファイル存在チェック ───────────────────────────────────
        if (!Files.exists(pathA)) {
            System.err.println("[ERROR] ファイルAが見つかりません: " + pathA.toAbsolutePath());
            System.exit(1);
        }
        if (!Files.exists(pathB)) {
            System.err.println("[ERROR] ファイルBが見つかりません: " + pathB.toAbsolutePath());
            System.exit(1);
        }

        // ── CSVファイル読み込み ────────────────────────────────────
        CsvData csvA;
        CsvData csvB;
        try {
            csvA = read(pathA);
            System.out.println("[INFO] ファイルAを読み込みました: " + pathA.getFileName()
                + "  (ヘッダー数=" + csvA.headers.size()
                + ", データ行数=" + csvA.rows.size() + ")");
        } catch (IOException e) {
            System.err.println("[ERROR] ファイルAの読み込みに失敗しました: " + e.getMessage());
            System.exit(1);
            return;
        }

        try {
            csvB = read(pathB);
            System.out.println("[INFO] ファイルBを読み込みました: " + pathB.getFileName()
                + "  (ヘッダー数=" + csvB.headers.size()
                + ", データ行数=" + csvB.rows.size() + ")");
        } catch (IOException e) {
            System.err.println("[ERROR] ファイルBの読み込みに失敗しました: " + e.getMessage());
            System.exit(1);
            return;
        }

        // ── マージ処理 ────────────────────────────────────────────
        MergeResult result = merge(csvA, csvB);

        if (!result.success) {
            System.err.println("[ERROR] マージに失敗しました。以下のヘッダーの不整合を確認してください:");
            for (String error : result.errors) {
                System.err.println("  - " + error);
            }
            System.err.println("[ERROR] " + OUTPUT_FILE_NAME + " は生成されませんでした。");
            System.exit(1);
            return;
        }

        // ── 出力 ─────────────────────────────────────────────────
        Path outputPath = Paths.get(OUTPUT_FILE_NAME);
        try {
            write(result.mergedData, outputPath);
            System.out.println("[INFO] マージ完了: " + outputPath.toAbsolutePath());
            System.out.println("[INFO] 総データ行数=" + result.mergedData.rows.size());
        } catch (IOException e) {
            System.err.println("[ERROR] " + OUTPUT_FILE_NAME + " の書き込みに失敗しました: " + e.getMessage());
            System.exit(1);
        }
    }

    // ═════════════════════════════════════════════════════════════
    // マージロジック
    // ═════════════════════════════════════════════════════════════

    /**
     * CSVデータAとBをマージする。
     *
     * @param csvA 基準となるCSVデータ
     * @param csvB 並び替え対象のCSVデータ
     * @return マージ結果（成功時はmergedData、失敗時はerrorsを持つ）
     */
    static MergeResult merge(CsvData csvA, CsvData csvB) {
        List<String> headersA = csvA.headers;

        // A・Bのヘッダーをlowercaseキーで管理（元の表記も保持）
        Map<String, String> aHeaderMap = new LinkedHashMap<>();
        for (String h : headersA) {
            aHeaderMap.put(h.toLowerCase(), h);
        }
        Map<String, String> bHeaderMap = new LinkedHashMap<>();
        for (String h : csvB.headers) {
            bHeaderMap.put(h.toLowerCase(), h);
        }

        // ── バリデーション: BにあってAにない列を検出 ───────────────
        List<String> errors = new ArrayList<>();
        for (Map.Entry<String, String> entry : bHeaderMap.entrySet()) {
            if (!aHeaderMap.containsKey(entry.getKey())) {
                errors.add(String.format(
                    "BにあってAにない列が見つかりました: \"%s\"", entry.getValue()));
            }
        }
        if (!errors.isEmpty()) {
            return MergeResult.failure(errors);
        }

        // ── Bの列インデックスマップ構築（lowercaseヘッダー → 列番号）──
        Map<String, Integer> bIndexMap = new LinkedHashMap<>();
        for (int i = 0; i < csvB.headers.size(); i++) {
            bIndexMap.put(csvB.headers.get(i).toLowerCase(), i);
        }

        // ── BのデータをAのヘッダー順に並び替え ──────────────────────
        List<List<String>> reorderedBRows = new ArrayList<>();
        for (List<String> bRow : csvB.rows) {
            List<String> reorderedRow = new ArrayList<>();
            for (String aHeader : headersA) {
                Integer bColIndex = bIndexMap.get(aHeader.toLowerCase());
                if (bColIndex != null && bColIndex < bRow.size()) {
                    reorderedRow.add(bRow.get(bColIndex));  // Bに対応列あり
                } else {
                    reorderedRow.add("");                   // AにあってBにない列 → 空欄
                }
            }
            reorderedBRows.add(reorderedRow);
        }

        // ── A + 並び替え後B を縦結合 ───────────────────────────────
        List<List<String>> mergedRows = new ArrayList<>();
        mergedRows.addAll(csvA.rows);
        mergedRows.addAll(reorderedBRows);

        return MergeResult.success(new CsvData(new ArrayList<>(headersA), mergedRows));
    }

    // ═════════════════════════════════════════════════════════════
    // CSV読み込み
    // ═════════════════════════════════════════════════════════════

    /**
     * 指定パスのCSVファイルを読み込む（UTF-8）。
     */
    static CsvData read(Path filePath) throws IOException {
        try (BufferedReader br = Files.newBufferedReader(filePath, StandardCharsets.UTF_8)) {
            String headerLine = br.readLine();
            if (headerLine == null) {
                throw new IOException("CSVファイルが空です: " + filePath);
            }
            List<String> headers = parseLine(headerLine);

            List<List<String>> rows = new ArrayList<>();
            String line;
            while ((line = br.readLine()) != null) {
                if (line.trim().isEmpty()) continue;  // 末尾の空行を無視
                rows.add(parseLine(line));
            }
            return new CsvData(headers, rows);
        }
    }

    /**
     * CSV1行をフィールドリストに分解する（RFC4180準拠）。
     * <ul>
     *   <li>ダブルクォート囲みフィールドに対応</li>
     *   <li>クォート内の "" は " 1文字にエスケープ</li>
     *   <li>クォートなしフィールドは前後スペースをトリム</li>
     * </ul>
     */
    static List<String> parseLine(String line) {
        List<String> fields = new ArrayList<>();
        int len = line.length();
        int i = 0;

        while (i <= len) {
            if (i < len && line.charAt(i) == '"') {
                // ── クォートありフィールド ──
                i++; // 開きクォートをスキップ
                StringBuilder sb = new StringBuilder();
                while (i < len) {
                    char c = line.charAt(i);
                    if (c == '"') {
                        if (i + 1 < len && line.charAt(i + 1) == '"') {
                            sb.append('"');  // "" → " エスケープ
                            i += 2;
                        } else {
                            i++;             // 閉じクォート
                            break;
                        }
                    } else {
                        sb.append(c);
                        i++;
                    }
                }
                fields.add(sb.toString());
                if (i < len && line.charAt(i) == ',') i++;
                else i = len + 1;
            } else {
                // ── クォートなしフィールド ──
                int commaPos = line.indexOf(',', i);
                if (commaPos == -1) {
                    fields.add(line.substring(i).trim());
                    i = len + 1;
                } else {
                    fields.add(line.substring(i, commaPos).trim());
                    i = commaPos + 1;
                }
            }
        }
        return fields;
    }

    // ═════════════════════════════════════════════════════════════
    // CSV書き出し
    // ═════════════════════════════════════════════════════════════

    /**
     * CsvDataを指定パスに書き出す（UTF-8、既存ファイルは上書き）。
     */
    static void write(CsvData csvData, Path outputPath) throws IOException {
        try (BufferedWriter bw = Files.newBufferedWriter(
                outputPath,
                StandardCharsets.UTF_8,
                StandardOpenOption.CREATE,
                StandardOpenOption.TRUNCATE_EXISTING)) {

            bw.write(toLine(csvData.headers));
            bw.newLine();
            for (List<String> row : csvData.rows) {
                bw.write(toLine(row));
                bw.newLine();
            }
        }
    }

    /**
     * フィールドリストを1行のCSV文字列に変換する。
     * カンマ・ダブルクォート・改行を含む値はダブルクォートで囲む。
     */
    static String toLine(List<String> fields) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < fields.size(); i++) {
            if (i > 0) sb.append(',');
            String field = fields.get(i);
            if (needsQuoting(field)) {
                sb.append('"').append(field.replace("\"", "\"\"")).append('"');
            } else {
                sb.append(field);
            }
        }
        return sb.toString();
    }

    static boolean needsQuoting(String field) {
        return field.contains(",") || field.contains("\"")
            || field.contains("\n") || field.contains("\r");
    }

    // ═════════════════════════════════════════════════════════════
    // 内部データクラス
    // ═════════════════════════════════════════════════════════════

    /** CSVデータ（ヘッダー行 + データ行）を保持する。 */
    static class CsvData {
        final List<String> headers;
        final List<List<String>> rows;

        CsvData(List<String> headers, List<List<String>> rows) {
            this.headers = headers;
            this.rows = rows;
        }
    }

    /** マージ結果（成功時はmergedData、失敗時はerrorsを持つ）。 */
    static class MergeResult {
        final boolean success;
        final CsvData mergedData;
        final List<String> errors;

        private MergeResult(boolean success, CsvData mergedData, List<String> errors) {
            this.success = success;
            this.mergedData = mergedData;
            this.errors = errors;
        }

        static MergeResult success(CsvData mergedData) {
            return new MergeResult(true, mergedData, new ArrayList<>());
        }

        static MergeResult failure(List<String> errors) {
            return new MergeResult(false, null, errors);
        }
    }
}
