@Component
@ServiceName(QueryService_.CDS_NAME)
public class QueryHandler implements EventHandler {

    private static final Logger log = LoggerFactory.getLogger(QueryHandler.class);

    private final RemoteService s4RemoteService;
    private final PersistenceService db;

    public QueryHandler(
            @Qualifier("S4RemoteService") RemoteService s4RemoteService,
            PersistenceService db) {
        this.s4RemoteService = s4RemoteService;
        this.db = db;
    }

    // -------------------------------------------------------
    // メインハンドラ
    // -------------------------------------------------------
    @On(event = CqnService.EVENT_READ, entity = MainEntity_.CDS_NAME)
    public void onReadMainEntity(CdsReadEventContext context) {

        // ② BTP結合データを取得
        List<BtpJoinView> btpJoinedList = fetchBtpJoinedData();
        if (btpJoinedList.isEmpty()) {
            context.setResult(Collections.emptyList());
            return;
        }

        // ① S4CDSViewAを取得（BTPデータのキーで絞り込んでS4へ問い合わせ）
        List<S4CDSViewA> viewAList = fetchS4CDSViewA(btpJoinedList);

        // ③ ViewAをBTPデータの条件でフィルター
        List<S4CDSViewA> filteredViewAList = filterViewAByBtpData(viewAList, btpJoinedList);

        // ④ S4CDSViewBを取得
        List<S4CDSViewB> viewBList = fetchS4CDSViewB(btpJoinedList);

        // ⑤ メインエンティティにマッピング
        List<MainEntity> results = buildResults(filteredViewAList, btpJoinedList, viewBList);

        context.setResult(results);
    }

    // -------------------------------------------------------
    // ② BTP結合ビューを取得
    // -------------------------------------------------------
    private List<BtpJoinView> fetchBtpJoinedData() {
        CqnSelect query = Select.from(BtpJoinView_.class);
        return db.run(query).listOf(BtpJoinView.class);
    }

    // -------------------------------------------------------
    // ① S4CDSViewAを取得
    // BTPデータのヘッダキー（会社コード＋受注NO）でS4に絞り込み問い合わせ
    // 全件取得後にJava側でフィルターすると取得量が膨大になるため、
    // S4へのリクエスト時点で可能な範囲を絞り込む
    // -------------------------------------------------------
    private List<S4CDSViewA> fetchS4CDSViewA(List<BtpJoinView> btpJoinedList) {

        CqnPredicate filter = btpJoinedList.stream()
                .map(b -> (CqnPredicate) CQL.and(
                        CQL.get(S4CDSViewA.COMPANY_CODE).eq(b.getHeaderCompanyCode()),
                        CQL.get(S4CDSViewA.ORDER_NO).eq(b.getHeaderOrderNo())
                ))
                .distinct()
                .reduce(CQL::or)
                .orElseThrow(() -> new IllegalStateException("BTPデータが空のため、S4CDSViewAのフィルター条件を構築できません"));

        CqnSelect query = Select.from(S4CDSViewA_.class).where(filter);
        return s4RemoteService.run(query).listOf(S4CDSViewA.class);
    }

    // -------------------------------------------------------
    // ③ ViewAをBTPデータの条件でフィルター
    //   ヘッダキー（会社コード＋受注NO）と
    //   明細キー（会社コード＋受注NO＋明細NO）の両方に一致するものだけ残す
    // -------------------------------------------------------
    private List<S4CDSViewA> filterViewAByBtpData(
            List<S4CDSViewA> viewAList,
            List<BtpJoinView> btpJoinedList) {

        // 検索コストをO(1)にするためSetに変換
        Set<String> headerKeySet = btpJoinedList.stream()
                .map(b -> toHeaderKey(b.getHeaderCompanyCode(), b.getHeaderOrderNo()))
                .collect(Collectors.toSet());

        Set<String> lineKeySet = btpJoinedList.stream()
                .map(b -> toLineKey(b.getLineCompanyCode(), b.getLineOrderNo(), b.getLineNo()))
                .collect(Collectors.toSet());

        return viewAList.stream()
                .filter(a -> headerKeySet.contains(toHeaderKey(a.getCompanyCode(), a.getOrderNo())))
                .filter(a -> lineKeySet.contains(toLineKey(a.getCompanyCode(), a.getOrderNo(), a.getLineNo())))
                .collect(Collectors.toList());
    }

    // -------------------------------------------------------
    // ④ S4CDSViewBを取得
    //   条件: 会社コード＝②ヘッダ.会社コード、出荷伝票番号＝②ヘッダ.受注NO
    // -------------------------------------------------------
    private List<S4CDSViewB> fetchS4CDSViewB(List<BtpJoinView> btpJoinedList) {

        CqnPredicate filter = btpJoinedList.stream()
                .map(b -> (CqnPredicate) CQL.and(
                        CQL.get(S4CDSViewB.COMPANY_CODE).eq(b.getHeaderCompanyCode()),
                        CQL.get(S4CDSViewB.DELIVERY_DOC_NO).eq(b.getHeaderOrderNo())
                ))
                .distinct()
                .reduce(CQL::or)
                .orElseThrow(() -> new IllegalStateException("BTPデータが空のため、S4CDSViewBのフィルター条件を構築できません"));

        CqnSelect query = Select.from(S4CDSViewB_.class).where(filter);
        return s4RemoteService.run(query).listOf(S4CDSViewB.class);
    }

    // -------------------------------------------------------
    // ⑤ メインエンティティを組み立て
    // -------------------------------------------------------
    private List<MainEntity> buildResults(
            List<S4CDSViewA> filteredViewAList,
            List<BtpJoinView> btpJoinedList,
            List<S4CDSViewB> viewBList) {

        // 検索効率化のためMapに変換
        Map<String, BtpJoinView> btpByLineKey = btpJoinedList.stream()
                .collect(Collectors.toMap(
                        b -> toLineKey(b.getLineCompanyCode(), b.getLineOrderNo(), b.getLineNo()),
                        Function.identity(),
                        (existing, duplicate) -> existing  // 重複時は先勝ち
                ));

        Map<String, S4CDSViewB> viewBByHeaderKey = viewBList.stream()
                .collect(Collectors.toMap(
                        b -> toHeaderKey(b.getCompanyCode(), b.getDeliveryDocNo()),
                        Function.identity(),
                        (existing, duplicate) -> existing
                ));

        return filteredViewAList.stream()
                .map(a -> toMainEntity(a, btpByLineKey, viewBByHeaderKey))
                .filter(Optional::isPresent)
                .map(Optional::get)
                .collect(Collectors.toList());
    }

    private Optional<MainEntity> toMainEntity(
            S4CDSViewA viewA,
            Map<String, BtpJoinView> btpByLineKey,
            Map<String, S4CDSViewB> viewBByHeaderKey) {

        String lineKey = toLineKey(viewA.getCompanyCode(), viewA.getOrderNo(), viewA.getLineNo());
        BtpJoinView btpData = btpByLineKey.get(lineKey);

        if (btpData == null) {
            log.warn("BTPデータが見つかりません。lineKey={}", lineKey);
            return Optional.empty();
        }

        String headerKey = toHeaderKey(btpData.getHeaderCompanyCode(), btpData.getHeaderOrderNo());
        S4CDSViewB viewB = viewBByHeaderKey.get(headerKey); // ViewBは存在しない場合もありうる

        MainEntity entity = MainEntity.create();

        // ⑤ マッピング定義（設計書の「例」に従う）
        entity.setCompanyCode(viewA.getCompanyCode());            // 会社コード ← ③(ViewA).会社コード
        entity.setOrderNo(btpData.getHeaderOrderNo());            // 受注NO ← ②(BTPヘッダ).受注NO
        entity.setLineNo(viewA.getLineNo());                      // 明細NO ← ③(ViewA).明細NO
        // entity.setSomeField(viewB != null ? viewB.getSomeField() : null);  // ViewBは null チェック必須

        return Optional.of(entity);
    }

    // -------------------------------------------------------
    // キー生成ユーティリティ
    // -------------------------------------------------------
    private static String toHeaderKey(String companyCode, String orderNo) {
        return companyCode + "|" + orderNo;
    }

    private static String toLineKey(String companyCode, String orderNo, String lineNo) {
        return companyCode + "|" + orderNo + "|" + lineNo;
    }
}


// -------------------------------------------------------
// ⑤ メインエンティティを組み立て（修正版）
// -------------------------------------------------------
private List<MainEntity> buildResults(
        List<S4CDSViewA> filteredViewAList,
        List<BtpJoinView> btpJoinedList,
        List<S4CDSViewB> viewBList) {

    // ✅ 修正: lineKeyのみ → ヘッダ＋明細の複合キーに変更
    //    これにより「このViewAレコードに対応するBTP行」を一意に特定できる
    Map<String, BtpJoinView> btpByCompositeKey = btpJoinedList.stream()
            .collect(Collectors.toMap(
                    b -> toBtpCompositeKey(
                            b.getHeaderCompanyCode(), b.getHeaderOrderNo(),
                            b.getLineCompanyCode(),   b.getLineOrderNo(), b.getLineNo()
                    ),
                    Function.identity(),
                    (existing, duplicate) -> {
                        // 同一複合キーの重複はデータ異常なのでWARNログを出す
                        log.warn("BTP複合キー重複検出。既存行を優先。key={}",
                                toBtpCompositeKey(
                                        existing.getHeaderCompanyCode(), existing.getHeaderOrderNo(),
                                        existing.getLineCompanyCode(), existing.getLineOrderNo(), existing.getLineNo()
                                ));
                        return existing;
                    }
            ));

    Map<String, S4CDSViewB> viewBByHeaderKey = viewBList.stream()
            .collect(Collectors.toMap(
                    b -> toHeaderKey(b.getCompanyCode(), b.getDeliveryDocNo()),
                    Function.identity(),
                    (existing, duplicate) -> existing
            ));

    return filteredViewAList.stream()
            .map(a -> toMainEntity(a, btpByCompositeKey, viewBByHeaderKey))
            .filter(Optional::isPresent)
            .map(Optional::get)
            .collect(Collectors.toList());
}

private Optional<MainEntity> toMainEntity(
        S4CDSViewA viewA,
        Map<String, BtpJoinView> btpByCompositeKey,
        Map<String, S4CDSViewB> viewBByHeaderKey) {

    // ✅ ③の5条件すべてを複合キーに反映してBTP行を特定
    //    headerOrderNo = viewA.orderNo かつ lineOrderNo = viewA.orderNo
    //    という整合した1行のみがヒットする
    String compositeKey = toBtpCompositeKey(
            viewA.getCompanyCode(), viewA.getOrderNo(),   // ヘッダ条件
            viewA.getCompanyCode(), viewA.getOrderNo(), viewA.getLineNo()  // 明細条件
    );

    BtpJoinView btpData = btpByCompositeKey.get(compositeKey);
    if (btpData == null) {
        log.warn("対応BTPデータなし。ViewAレコードをスキップ。compositeKey={}", compositeKey);
        return Optional.empty();
    }

    // ✅ btpDataは確実に正しいヘッダ情報を持つため、ViewBルックアップも正確
    String headerKey = toHeaderKey(btpData.getHeaderCompanyCode(), btpData.getHeaderOrderNo());
    S4CDSViewB viewB = viewBByHeaderKey.get(headerKey);

    MainEntity entity = MainEntity.create();
    entity.setCompanyCode(viewA.getCompanyCode());
    entity.setOrderNo(btpData.getHeaderOrderNo());
    entity.setLineNo(viewA.getLineNo());
    // ... 他フィールド

    return Optional.of(entity);
}

// -------------------------------------------------------
// キー生成ユーティリティ
// -------------------------------------------------------
private static String toHeaderKey(String companyCode, String orderNo) {
    return companyCode + "|" + orderNo;
}

private static String toBtpCompositeKey(
        String headerCompanyCode, String headerOrderNo,
        String lineCompanyCode,   String lineOrderNo, String lineNo) {
    return headerCompanyCode + "|" + headerOrderNo
            + "|" + lineCompanyCode + "|" + lineOrderNo + "|" + lineNo;
}
